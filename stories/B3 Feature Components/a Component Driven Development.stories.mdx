# Component-Driven Development with storybook

A bunch of content

```tsx
// UserList.stories.ts|tsx

import React from "react";

import { ComponentStory, ComponentMeta } from "@storybook/react";

import { ApolloProviderWithInterceptor } from "common/tests/utils/renderWithApolloInterceptor";
import { UserList } from "./UserList";

export default {
  /* CDD Components should be added to the "Feature Components" section as they are not part of the Design System */
  title: "Feature Components/UserList",
  component: UserList,
  decorators: [
    // We must wrap the story in our mock `ApolloProviderWithInterceptor`
    (Story) => (
      <ApolloProviderWithInterceptor>
        <Story />
      </ApolloProviderWithInterceptor>
    ),
  ],
} as ComponentMeta<typeof UserList>;

export const SuccessUserList: ComponentStory<typeof UserList> = () => <UserList />;
```
